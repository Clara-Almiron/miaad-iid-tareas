// Modelo analitico final derivado desde la tabla raw unica newsapi.everything en MotherDuck.
// Para esta tarea academica, las claves derivadas se modelan como uint64.

Table dim_article {
  article_sk uint64 [pk]
  url varchar
  title varchar
  airbyte_raw_id varchar
}

Table dim_source {
  source_sk uint64 [pk]
  source_api_id varchar
  source_name varchar
}

Table dim_author {
  author_sk uint64 [pk]
  author_name varchar
}

Table dim_published_datetime {
  published_datetime_sk uint64 [pk]
  published_at timestamp
  published_date date
  published_year int
  published_month int
  published_day int
  published_hour int
  day_of_week varchar
}

Table fct_news_articles {
  article_event_sk uint64 [pk]
  article_sk uint64 [not null, ref: > dim_article.article_sk]
  source_sk uint64 [not null, ref: > dim_source.source_sk]
  author_sk uint64 [ref: > dim_author.author_sk]
  published_datetime_sk uint64 [ref: > dim_published_datetime.published_datetime_sk]

  article_count int
  airbyte_extracted_at timestamp
}
